	sir = f(A,B,C,D)
	sendo:
	A = valor anterior da sirene;
	B = estado do alarme;
	C = sensor;
	D = tempo.
	sir = ABCD + ABC + ABD + AB + BD <=> sir = AB + BD <=> sir = B (A + D)

	r = f(A,B,C,D)
	sendo:
	A = valor anterior da sirene;
	B = estado do alarme;
	C = sensor;
	D = tempo.
	r = B * ~(ACD)
